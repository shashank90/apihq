# Install postgres

Pre-requisite for installing psycopg2
# sudo apt-get install build-essential
# sudo apt install python3-dev libpq-dev
# Install node and npm
https://www.digitalocean.com/community/tutorials/how-to-install-node-js-on-ubuntu-20-04

Install python-venv
sudo apt install python3.8-venv
python3 -m venv venv
sudo apt-get install python3-pip
sudo apt-get install libpq-dev python-dev
pip install -r requirements.txt


# Install IBM OpenAPI validator ( I guess for these to work, they need to be installed inside venv. Try installing globally and check)
sudo npm install -g ibm-openapi-validator
sudo npm install -g @ibm-cloud/openapi-ruleset

# Install openapi-code generator

sudo apt install default-jdk
sudo apt install default-jre
sudo apt install maven
sudo apt install jq

# Copy paste following instructions into terminal

mkdir -p ~/bin/openapitools
curl https://raw.githubusercontent.com/OpenAPITools/openapi-generator/master/bin/utils/openapi-generator-cli.sh > ~/bin/openapitools/openapi-generator-cli
chmod u+x ~/bin/openapitools/openapi-generator-cli
export PATH=$PATH:~/bin/openapitools/

sudo apt install ngnix
